
CREATE TABLE telcodb.dbo.support_menu (
	menu_id numeric identity NOT NULL,
	menu_nm varchar(25) NULL,
	created_dt datetime NULL,
	createdBy int NOT NULL,
	menu_comments varchar(100) NOT NULL,
	PRIMARY KEY (menu_id)
);

CREATE TABLE telcodb.dbo.support_menuitem (
	menuitem_id numeric identity NOT NULL,
	menuitem_nm varchar(100) NOT NULL,
	menuitem_link varchar(100) NOT NULL,
	menuitem_comments varchar(100) NOT NULL,
	createdBy int NOT NULL,
	created_dt datetime NOT NULL,
	extraParam varchar(100) NOT NULL,
	PRIMARY KEY (menuitem_id)
);

CREATE TABLE telcodb.dbo.support_menuitemToIPAddress (
	id numeric identity NOT NULL,
	menuitem_id int NOT NULL,
	ipaddress_id int NOT NULL,
	createdBy int NOT NULL,
	created_dt datetime NOT NULL,
	PRIMARY KEY (id)
);

CREATE TABLE telcodb.dbo.support_menuitemToMenu (
	id numeric identity NOT NULL,
	menuitem_id int NOT NULL,
	menu_id int NOT NULL,
	createdBy int NOT NULL,
	created_dt datetime NOT NULL,
	PRIMARY KEY (id)
);

CREATE TABLE telcodb.dbo.support_menuitemToRole (
	id numeric identity NOT NULL,
	menuitemToMenu_id int NOT NULL,
	userType_id int NOT NULL,
	createdBy int NOT NULL,
	created_dt datetime NOT NULL,
	PRIMARY KEY (id)
);

CREATE TABLE telcodb.dbo.support_menuToRole (
	id numeric identity NOT NULL,
	menu_id int NOT NULL,
	userType_id int NOT NULL,
	createdBy int NOT NULL,
	created_dt datetime NOT NULL,
	PRIMARY KEY (id)
);
